



CREATE OR REPLACE PROCEDURE proc_Grade(
 p_roll IN NUMBER,
 p_name IN VARCHAR2,
p_marks IN NUMBER
 )
 AS
 BEGIN
 IF p_marks BETWEEN 990 AND 1500 THEN
 INSERT INTO RESULT VALUES(p_roll,p_name,'DISTINCTION');
 
 ELSIF p_marks BETWEEN 990 AND 989 THEN
  INSERT INTO RESULT VALUES(p_roll,p_name,'FIRST CLASS');
 
 ELSIF p_marks BETWEEN 825 AND 899 THEN
  INSERT INTO RESULT VALUES(p_roll,p_name,'HIGHER SECONDARY CLASS');
 
 ELSE
 INSERT INTO RESULT VALUES(p_roll,p_name,'FAIL');
 END IF;
 
 INSERT INTO stud1_marks VALUES(p_roll,p_name,p_marks);
   END;
